    private Trie searchPrefix(String prefix) {
        Trie node = this;

        for (int i = 0; i < prefix.length(); i++) {
            char ch = prefix.charAt(i);
            int index = ch - 'a';

            if (node.children[index] == null)
                return null;

            node = node.children[index];
        }
        return node;

class Solution {
        // Â≠óÂÖ∏Ê†ëÁöÑÊ†πËäÇÁÇπ
        Trie root = new Trie();

        static final int HIGH_BIT = 30;

        public int findMaximumXOR(int[] nums) {
            int n = nums.length;
            int x = 0;
            for (int i = 1; i < n; ++i) {
                // Â∞Ü nums[i-1] ÊîæÂÖ•Â≠óÂÖ∏Ê†ëÔºåÊ≠§Êó∂ nums[0 .. i-1] ÈÉΩÂú®Â≠óÂÖ∏Ê†ë‰∏≠

                x = Math.max(x, check(nums[i]));
            }
            return x;
        }

        public void add(int num) {
            Trie cur = root;
            for (int k = HIGH_BIT; k >= 0; --k) {
                int bit = (num >> k) & 1;
                if (bit == 0) {
                    if (cur.left == null) {
                        cur.left = new Trie();
                    }
                    cur = cur.left;
                } else {
                    if (cur.right == null) {
                        cur.right = new Trie();
                    }
                    cur = cur.right;
                }
            }
        }

        public int check(int num) {
            Trie cur = root;
            int x = 0;
            for (int k = HIGH_BIT; k >= 0; --k) {
                int bit = (num >> k) & 1;
                if (bit == 0) {
                    // a_i ÁöÑÁ¨¨ k ‰∏™‰∫åËøõÂà∂‰Ωç‰∏∫ 0ÔºåÂ∫îÂΩìÂæÄË°®Á§∫ 1 ÁöÑÂ≠êËäÇÁÇπ right Ëµ∞
                    if (cur.right != null) {
                        cur = cur.right;
                        x = x * 2 + 1;
                    } else {
                        cur = cur.left;
                        x = x * 2;
                    }
                } else {
                    // a_i ÁöÑÁ¨¨ k ‰∏™‰∫åËøõÂà∂‰Ωç‰∏∫ 1ÔºåÂ∫îÂΩìÂæÄË°®Á§∫ 0 ÁöÑÂ≠êËäÇÁÇπ left Ëµ∞
                    if (cur.left != null) {
                        cur = cur.left;
                        x = x * 2 + 1;
                    } else {
                        cur = cur.right;
                        x = x * 2;
                    }
                }
            }
            return x;
        }
    }
