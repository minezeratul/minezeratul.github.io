    public int deleteAndEarn(int[] nums){
        int n = nums.length;
        if (n == 0 || nums == null)
            return 0;
        else if (n == 1)
            return nums[0];

        int max = 0;
        for (int maxVal:nums) {
            max = Math.max(maxVal , max);//Èò≤Ê≠¢out of bounds
        }

        int[] all = new int[max + 1];
        for (int val : nums)
            all[val] += val;

        return rob(all);
    }

    private int rob(int[] nums){
        int n = nums.length;

        int first = nums[0] , second = Math.max(nums[0] , nums[1]);
        for (int i = 2; i < n; i++) {
            int temp = second;
            second = Math.max(first + nums[i], second);
            first = temp;
        }

        return second;
    }
