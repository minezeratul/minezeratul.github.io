public void solve(char[][] board) {
        if (board.length == 0){
            return;
        }

        int m = board.length;
        int n = board[0].length;
        UnionFind uf = new UnionFind(m * n + 1);

                uf.union(i * n , dummy);

            if (board[i][n - 1] == 'O'){
                uf.union(i * n + n - 1 , dummy);

                uf.union(n * (m - 1) + j, dummy);

                        int y = j + direction[k][1];
